Sử dụng


						npm install --save express
						# hoặc
						yarn add express
					

						const express = require('express');
						const app = express();
						const port = 3000;

						app.get('/', (req, res) => res.send('Hello World!'));

						app.listen(port, () => console.log(`Example app listening on port ${port}!`));

Routing


						app.get();
